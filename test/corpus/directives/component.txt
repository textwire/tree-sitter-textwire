==================
Complex nested directives
==================
@component('list')
    @each(item in items)
        @if(item.active)
            <span>{{ item.name }}</span>
        @end
    @end
@end
---

(program
    (comp_dir
        name: (str_expr)
        block: (block
            (text)
            (each_dir
                var: (ident_expr)
                array: (ident_expr)
                block: (block
                    (text)
                    (if_dir
                        cond: (dot_expr
                            left: (ident_expr)
                            key: (ident_expr))
                        if_block: (block
                            (text)
                            (embedded
                                (dot_expr
                                    left: (ident_expr)
                                    key: (ident_expr)))
                            (text)))
                    (text)))
            (text))))

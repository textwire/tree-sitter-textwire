==================
Inline @component without text
==================
@component('user')
---

(program
    (comp_dir
        name: (str_expr)))

==================
Inline @component with short object notation
==================
@component('book', { title })
---

(program
    (comp_dir
        name: (str_expr)
        argument: (obj_expr
            (pairs
                (pair
                    key: (ident_expr))))))

==================
Inline @component with full object notation

---

(program
    (comp_dir

        argument: (obj_expr
            (pairs
                (pair
                    key: (ident_expr)
                    val: (str_expr))))))

==================
Block @component no arguments
==================
@component('book')
    @slot<div>Nice</div>@end
@end
---

(program
    (comp_dir
        name: (str_expr)
        block: (block
            (text)
            (slot_dir
                block: (block (text)))
            (text))))

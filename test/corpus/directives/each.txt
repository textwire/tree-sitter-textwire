==================
Each statement without any block
==================
@each(name in names)@end
---

(program 
    (each_statement
        var: (identifier)
        array: (identifier)))

==================
Inline each statement with 1 HTML statement
==================
@each(name in [])<span>Nice</span>@end
---

(program 
    (each_statement
        var: (identifier)
        array: (array_literal)
        block: (block_statement
            (text))))

==================
Multiline each statement with expression
==================
@each(name in names)
    <span>{{ value }}</span>
@end
---

(program 
    (each_statement
        var: (identifier)
        array: (identifier)
        block: (block_statement
            (text)
            (brace_statement
                (statement
                    (expression_statement
                        (identifier))))
            (text))))

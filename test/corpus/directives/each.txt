==================
Each statement without any block
==================
@each(name in names)@end
---

(program 
    (each_dir
        var: (ident_expr)
        array: (ident_expr)))

==================
Inline each statement with 1 HTML statement
==================
@each(name in [])<span>Nice</span>@end
---

(program 
    (each_dir
        var: (ident_expr)
        array: (arr_expr)
        block: (block
            (text))))

==================
Multiline each statement with expression
==================
@each(name in names)
    <span>{{ value }}</span>
@end
---

(program 
    (each_dir
        var: (ident_expr)
        array: (ident_expr)
        block: (block
            (text)
            (embedded
                (ident_expr))
            (text))))
